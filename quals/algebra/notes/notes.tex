\subsection{Subgroups and Quotient Groups}

\begin{definition}
  Let $G$ be a group and $H$ a nonempty subset that is closed under the product in $G$. If $H$ itself is a group under 
  the product in $G$, then $H$ is said to be a \textbf{subgroup} of $G$. This is denoted by $H \subset G$. 
\end{definition}

\begin{definition}
  A subgroup other than the group itself and the trivial subgroup $\langle e \rangle$ is called a \textbf{proper subgroup}
\end{definition}

\begin{theorem}{}
  Let $H$ be a nonempty subset of a group $G$. Then $H$ is a subgroup of $G$ if and only if $ab^{-1} \in H$ for all $a,b \in H$. 
\end{theorem}
